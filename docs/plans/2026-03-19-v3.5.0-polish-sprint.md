## Task 2: Fix GitHub #4 — Import Shows Wrong Active Chart

**Files:**
- Modify: `src/editor/chart-editor.ts:64-82`
- Test: `tests/editor/chart-editor.test.ts`

**Bug Summary:** When importing as a new chart, `chartStore.createChartFromTree()` sets `activeChartId` and emits. ChartEditor's `onChange` listener triggers `refresh()` asynchronously. But `refresh()` has a queuing mechanism (lines 67-81) — if a refresh is already in progress, the new one gets queued. The chart list may render with stale data if the `onComplete` callback in main.ts fires before ChartEditor's async refresh resolves.

**Root Cause:** The `createChartFromTree()` call in `src/main.ts:314` fires `emit()` which triggers ChartEditor.refresh() asynchronously, but the rest of the onComplete callback (lines 315-325) continues immediately. The chart header updates to the new chart, but the sidebar chart list hasn't refreshed yet — showing the previous chart as "active" until the async refresh completes.

**Step 1: Write failing test**

Add to `tests/editor/chart-editor.test.ts`:

```typescript
it('shows newly imported chart as active after createChartFromTree', async () => {
  // Create initial chart
  await chartStore.init();
  const initial = chartStore.getActiveChart()!;

  // Simulate what main.ts does on import: create a new chart from tree
  const newTree: OrgNode = {
    id: 'imported-root',
    name: 'Imported CEO',
    title: 'CEO',
    children: [],
  };
  const newChart = await chartStore.createChartFromTree('Imported Org', newTree);

  // Wait for ChartEditor's async refresh to complete
  await vi.waitFor(() => {
    const activeEl = container.querySelector('.chart-item.active');
    expect(activeEl).not.toBeNull();
    expect(activeEl!.textContent).toContain('Imported Org');
  });

  // The previously active chart should NOT be marked active
  const allActive = container.querySelectorAll('.chart-item.active');
  expect(allActive.length).toBe(1);
});
```

**Step 2: Run test to verify it fails**

```bash
npx vitest run tests/editor/chart-editor.test.ts -t "shows newly imported chart as active"
```

Expected: FAIL — the async refresh may not have completed, or the active class may be on the wrong chart.

**Step 3: Implement the fix**

The fix is to expose a way for main.ts to explicitly await ChartEditor's refresh after creating a chart. Modify `chart-editor.ts`:

```typescript
// In ChartEditor class, add a public method:
async awaitRefresh(): Promise<void> {
  // If a refresh is in progress, wait for it plus any queued refresh
  while (this.refreshInProgress || this.refreshQueued) {
    await new Promise(resolve => setTimeout(resolve, 10));
  }
}
```

Then in `src/main.ts`, after `createChartFromTree` in the onComplete handler (~line 314), add:

```typescript
const chart = await chartStore.createChartFromTree(wizardState.chartName, finalTree);
await chartEditor.awaitRefresh(); // Wait for sidebar to update
```

**Important:** `chartEditor` must be accessible in the `onComplete` closure. Check that it's declared before the importWizard setup. If not, restructure to pass it.

**Alternative simpler fix:** Make `ChartEditor.refresh()` return its Promise and have ChartStore emit await the listeners. But this would change the store's event model. The simpler approach: in main.ts, after `createChartFromTree`, explicitly call `await chartEditor.refresh()` to force a synchronous update:

```typescript
// In main.ts onComplete callback, after line 314:
const chart = await chartStore.createChartFromTree(wizardState.chartName, finalTree);
// Force the chart editor to refresh synchronously before continuing
await chartEditor.refresh();
```

This works because `chartEditor.refresh()` already returns a Promise. The onChange listener will also fire, but the queuing mechanism handles that gracefully.

```typescript
// tests/helpers/tree-generator.ts
import type { OrgNode } from '../../src/types';
import { generateId } from '../../src/utils/id';

/**
 * Generate a synthetic org tree with the specified number of nodes.
 * Creates a balanced tree with ~3-5 children per manager.
 */
export function generateTree(nodeCount: number): OrgNode {
  const root: OrgNode = {
    id: generateId(),
    name: 'CEO',
    title: 'Chief Executive Officer',
    children: [],
  };

  let created = 1;
  const queue: OrgNode[] = [root];

  while (created < nodeCount && queue.length > 0) {
    const parent = queue.shift()!;
    const childCount = Math.min(
      Math.floor(Math.random() * 3) + 3, // 3-5 children
      nodeCount - created,
    );
    parent.children = [];
    for (let i = 0; i < childCount; i++) {
      const child: OrgNode = {
        id: generateId(),
        name: `Person ${created + i}`,
        title: `Title ${created + i}`,
        children: [],
      };
      parent.children.push(child);
      queue.push(child);
    }
    created += childCount;
  }

  return root;
}
```

```typescript
// tests/renderer/performance.test.ts
import { describe, it, expect, beforeEach } from 'vitest';
import { ChartRenderer } from '../../src/renderer/chart-renderer';
import { generateTree } from '../helpers/tree-generator';
import { defaultRendererOptions } from '../../src/store/settings-store';

describe('rendering performance', () => {
  let container: HTMLDivElement;
  let opts: any;

  beforeEach(() => {
    container = document.createElement('div');
    document.body.appendChild(container);
    opts = { ...defaultRendererOptions(), container, categories: [] };
  });

  it('renders 100-node tree without errors', () => {
    const tree = generateTree(100);
    const renderer = new ChartRenderer(opts);
    expect(() => renderer.render(tree)).not.toThrow();
  });

  it('renders 500-node tree in under 2 seconds', () => {
    const tree = generateTree(500);
    const renderer = new ChartRenderer(opts);
    const start = performance.now();
    renderer.render(tree);
    const duration = performance.now() - start;
    expect(duration).toBeLessThan(2000);
  });

  it('renders 1000-node tree in under 5 seconds', () => {
    const tree = generateTree(1000);
    const renderer = new ChartRenderer(opts);
    const start = performance.now();
    renderer.render(tree);
    const duration = performance.now() - start;
    expect(duration).toBeLessThan(5000);
  });

  it('re-renders do not leak DOM nodes', () => {
    const tree = generateTree(50);
    const renderer = new ChartRenderer(opts);
    renderer.render(tree);
    const countAfterFirst = container.querySelectorAll('*').length;
    renderer.render(tree);
    const countAfterSecond = container.querySelectorAll('*').length;
    // DOM node count should be approximately the same after re-render
    expect(Math.abs(countAfterSecond - countAfterFirst)).toBeLessThan(5);
  });
});
```
